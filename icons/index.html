#图标库

LynKit Icons 提供了 1000+ 精美图标，支持按需加载，体积小巧。

#安装

1pnpm add @lynkit/icons

#基础用法

图标组件采用 Icon + 图标名称（小写） 的命名规范。

1import { Iconhome, Iconsearch, Iconuser } from '@lynkit/icons';
2
3function App() {
4  return (
5    <div style={{ display: 'flex', gap: '16px' }}>
6      <Iconhome />
7      <Iconsearch />
8      <Iconuser />
9    </div>
10  );
11}
12
13export default App;

#自定义样式

支持通过 size、color、style 等属性自定义图标样式。

1import { Iconhome, Iconsetting, Iconnotice } from '@lynkit/icons';
2
3function App() {
4  return (
5    <div style={{ display: 'flex', gap: '16px', alignItems: 'center' }}>
6      <Iconhome size={24} color="#4f46e5" />
7      <Iconsetting size={32} color="#059669" />
8      <Iconnotice size={40} color="#dc2626" />
9    </div>
10  );
11}
12
13export default App;

#常用图标示例

1import {
2  Iconhome,
3  Iconsearch,
4  Iconuser,
5  Iconsetting,
6  Iconnotice,
7  Iconlike,
8  Iconedit,
9  Icondelete,
10  Icondownload,
11  Iconupload,
12} from '@lynkit/icons';
13
14function IconGrid() {
15  const icons = [
16    { Icon: Iconhome, name: 'home' },
17    { Icon: Iconsearch, name: 'search' },
18    { Icon: Iconuser, name: 'user' },
19    { Icon: Iconsetting, name: 'setting' },
20    { Icon: Iconnotice, name: 'notice' },
21    { Icon: Iconlike, name: 'like' },
22    { Icon: Iconedit, name: 'edit' },
23    { Icon: Icondelete, name: 'delete' },
24    { Icon: Icondownload, name: 'download' },
25    { Icon: Iconupload, name: 'upload' },
26  ];
27
28  return (
29    <div style={{
30      display: 'grid',
31      gridTemplateColumns: 'repeat(5, 1fr)',
32      gap: '24px'
33    }}>
34      {icons.map(({ Icon, name }) => (
35        <div
36          key={name}
37          style={{
38            display: 'flex',
39            flexDirection: 'column',
40            alignItems: 'center',
41            gap: '8px'
42          }}
43        >
44          <Icon size={32} />
45          <span style={{ fontSize: '12px', color: '#666' }}>{name}</span>
46        </div>
47      ))}
48    </div>
49  );
50}
51
52export default IconGrid;

#API

#Props

属性说明类型默认值
size图标大小number | string'1em'
color图标颜色string'currentColor'
className自定义类名string-
style自定义样式CSSProperties-

#命名规范

图标命名遵循以下规范：

类型格式示例
基础图标Icon{name}Iconhome, Iconsearch
填充变体Icon{name}fillIconhomefill, Iconuserfill
静态变体Icon{name}staticIconhomestatic

#按需加载

所有图标都支持按需加载，只需导入所需的图标组件即可：

1// ✅ 推荐：按需导入
2import { Iconhome, Iconsearch } from '@lynkit/icons';
3
4// ❌ 不推荐：全量导入会增加打包体积
5import * as Icons from '@lynkit/icons';
